//Crear un nuevo repuesto.
POST /repuestos
  - Request Body: (permite omitir el campo anio y foto)
    {
      "nombre": "string",
      "marca": "string",
      "modelo": "string",
      "anio": number,
      "precio": number,
      "foto": "string"
    }
  - Response:
    {
      "id": number,
      "nombre": "string",
      "marca": "string",
      "modelo": "string",
      "anio": number,
      "precio": number,
      "foto": "string"
    }

// Obtener todos los repuestos
GET /repuestos
  - Response:
    [
      {
        "id": number,
        "nombre": "string",
        "marca": "string",
        "modelo": "string",
        "anio": number,
        "precio": number,
        "foto": "string"
      },
      ...
    ]

      "modelo": "string",
      "anio": number,
      "precio": number,
      "foto": "string"
    }
